title: 'List Preview URLs'
api: 'GET /api/sandboxes/{id}/preview'

</ParamField>

```json 200
    "port": 3000,
    "sslStatus": "active",
    "createdAt": "2025-01-15T10:30:00Z"
  }
]
```
</ResponseExample>
